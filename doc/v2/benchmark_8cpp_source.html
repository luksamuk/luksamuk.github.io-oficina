36 void oficina::ofBenchmarkUpdateCall() {
37  if(!benchmarkStarted) return;
38  frameCount++;
39  if(benchmarkTime.yieldSpan() >= spanTime) {
40  ofLog(ofLogInfo, OFLOG_MAG "BENCHMARK:" OFLOG_RESET
41  " %0.2f seconds: %0.2f FPS, %f ms/frame\n",
42  benchmarkTime.yieldSpan(),
43  (frameCount / benchmarkTime.yieldSpan()),
44  (benchmarkTime.yieldSpan() / frameCount));
45  frameCount = 0u;
46  benchmarkTime.resetSpan();
47  }
