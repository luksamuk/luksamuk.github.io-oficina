29  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
30  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);

32  SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, 8);
33  SDL_GL_SetAttribute(SDL_GL_RED_SIZE, 8);

39  SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
40  SDL_GL_SetAttribute(SDL_GL_SHARE_WITH_CURRENT_CONTEXT, 1);
41 
42  m_wnd = SDL_CreateWindow(m_title.c_str(),
43  SDL_WINDOWPOS_CENTERED,
44  SDL_WINDOWPOS_CENTERED,

46  SDL_WINDOW_OPENGL | SDL_WINDOW_SHOWN);
