38 void oficina::ofScmInit() {
39  if(g_scm) {
40  ofLog(ofLogErr, "ofScmInit: Cannot re-initialize Scheme\n");
41  return;
42  }
43  g_scm = new scheme;
44  if(!scheme_init(g_scm)) {
45  ofLog(ofLogCrit, "ofScmInit: Couldn't initialize Scheme.\n");
46  g_scm = nullptr;
47  return;
48  }
49 
50  g_scm->interactive_repl = 1;
51 
52  // Bind error output
53  ofScmResetOutput();
54 
55  // Load default bindings
56  scheme_load_string(g_scm, ofScmInitSrc);
57 
58  // Clear debugger repl
59  ofCanvasManager::dbg_ReplOutStream().str("");
60 
61  // Define default functions
62  ofscm_defcommon(g_scm);
63 
64  // Define object api
65  ofscm_defobjapi(g_scm);
66 
67  // Small scheme test
68  scheme_load_string(g_scm,
69  "(display \"TinyScheme v1.41 on OficinaFramework\nPress F2 to toggle input.\n\")");
70 }

132 void ofScheme::init(std::string objname, ofEntity* e_this)
133 {
134  if(scm) {
135  ofLog(ofLogErr, "ofScheme.init: Cannot re-initialize Scheme\n");
136  return;
137  }
138  scm = new scheme;
139  if(!scheme_init(scm)) {
140  ofLog(ofLogCrit, "ofScheme.init: Couldn't initialize Scheme.\n");
141  scm = nullptr;
142  return;
143  }
144 
145  entityptr = e_this;
146 
147  scm->interactive_repl = 1;
148  scheme_set_output_port_string(scm, error_buffer, error_buffer + 255);
149 
150  // Load default bindings
151  scheme_load_string(scm, ofScmInitSrc);
152 
153  // Define default functions
154  ofscm_defcommon(scm);
155 
156  /* OBJECT ACCESS API */
157  //+this+ ; Retrieves reference to this object
158  scm->vptr->scheme_define(scm,
159  scm->global_env,
160  scm->vptr->mk_symbol(scm, "+this+"),
161  scm->vptr->mk_integer(scm, reinterpret_cast<long>(e_this)));
162 
163  // Define object-related API
164  ofscm_defobjapi(scm);
165 
166  // Add object to hash
167  std::stringstream oss;
168  ofdword instance = 0u;
169  auto iter = objMap.end();
170  do {
171  oss.str("");
172  oss << objname << "#" << instance;
173  iter = objMap.find(oss.str());
174 
175  } while(iter != objMap.end());
176 
177  e_this->setName(oss.str());
178  objMap[oss.str()] = e_this;
179 
180  m_initialized = true;
181 }

183 void ofScheme::load(std::string filename)
184 {
185  if(!m_initialized) {
186  ofLog(ofLogErr, "ofScheme.load: Can't load on unitialized Scheme.\n");
187  return;
188  }
189 
190  if(m_loaded) {
191  ofLog(ofLogWarn, "ofScheme.load: Reloading symbols for \"%s\"\n", filename.c_str());
192  }
193 
194  FILE* fp;
195  fp = fopen(filename.c_str(), "r");
196 
197  if(!fp) {
198  ofLog(ofLogErr, "ofScheme.load: Couldn't load script file \"%s\".\n", filename.c_str());
199  return;
200  }
201  scheme_load_file(scm, fp);
202  scheme_load_string(scm, "(init)");
203  fclose(fp);
204 
205  m_loaded = true;
206 }

220 void ofScheme::update(float dt)
221 {
222  if(!m_initialized && !m_loaded)
223  return;
224  if(error_buffer[0] != '\0') {
225  ofLog(ofLogErr, OFLOG_CYN "SCM: " OFLOG_RESET "%s", error_buffer);

228  }
