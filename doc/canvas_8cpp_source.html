49 void ofCanvasManager::update(float dt) {
50  bool addedNewCanvas = false;
51  while(!canvasQueue.empty()) {
52  addedNewCanvas = true;
53 
54  // TODO: Init on separate thread
55  canvasQueue.front()->init();
56  canvasQueue.front()->m_init = true;
57  canvasQueue.front()->load();
58  canvasQueue.front()->m_load = true;
59 
60  canvasList.push_back(canvasQueue.front());
61  canvasQueue.pop();
62  }
63  if(addedNewCanvas) {
64  canvasList.sort(
65  [](const oficina::ofCanvas* fst, const oficina::ofCanvas* snd) {
66  return fst->depth < snd->depth;
67  });
68  }
69 
70  m_debugger.update(dt);
71  for(auto canvas : canvasList) {
72  canvas->update(dt);
73  }
74 }

183 void ofCanvasManager::ofDebugCanvas::update(float dt)
184 {
185  if(!m_state) return;
186 
187  glm::uvec2 vwp = ofGetViewportSize(),
188  wnd = ofGetWindowSize();
189 
190  // Verify orthographic projection
191  if(debugViewport.x != wnd.x ||
192  debugViewport.y != wnd.y) {
193  // Recalculate MVP
194  debugViewport = wnd;
195  debugMVP = glm::mat4();
196  debugMVP =
197  glm::ortho(
198  0.0f,
199  float(debugViewport.x),
200  -float(debugViewport.y),
201  0.0f, 1.0f, 10.0f) *
202  glm::lookAt(
203  glm::vec3(0.0f, 0.0f, -1.2f),
204  glm::vec3(0.0f, 0.0f, 0.0f),
205  glm::vec3(0.0f, -1.0f, 0.0f));
206  }
207 
208  if(m_state == 1u) {
209  // Clear immediate mode debugger
210  debugStream.str("");
211 
212  if(debugFPSGetSpan.yieldSpan() >= 0.5f) {
213  debugFPS = (1.0f / dt);
214  debugFPSGetSpan.resetSpan();
215  }
216 
217  debugStream << " Debugger " << std::endl
218  << "==============" << std::endl
219  << "EVR >> " << OF_VERSION_STRING << std::endl
220  << "INP >> " << "P1[" << (ofIsGamepadConnected(ofPlayerOne) ? "G" : "K") << "] "
221  << "P2[" << (ofIsGamepadConnected(ofPlayerTwo) ? "G" : "K") << "] "
222  << "P3[" << (ofIsGamepadConnected(ofPlayerThree) ? "G" : "K") << "] "
223  << "P4[" << (ofIsGamepadConnected(ofPlayerFour) ? "G" : "K") << "]"
224  << std::endl
225  << "WND >> " << wnd.x << "x" << wnd.y << std::endl
226  << "VWP >> " << vwp.x << "x" << vwp.y << std::endl
227  << "FPS >> " << debugFPS << std::endl;
228  }
229  else if(m_state == 2u) {
230  if(m_shouldEval) {
231  if(ofGetTextInput() != "") {
232  debugReplHistory.push_back(ofGetTextInput());
233  ofScmEval(ofGetTextInput());
234  }
235  debugReplHistoryLast = debugReplHistory.end();
236 
237  if(ofScmGetOutputPtr()[0] != '\0') {
238  debugReplAnsStream << '[' << std::setfill('0') << std::setw(3)
239  << debugReplLinum << ']' << ' '
240  << ofScmGetOutputPtr();
241  //ofScmGetOutputPtr()[0] = '\0';
242  ofScmResetOutput();
243  }
244 
245  debugReplLinum++;
246  m_shouldEval = false;
247  ofClearTextInput();
248  //ofStartTextInput();
249  }
250  if(caretTimer.yieldSpan() >= 0.5f) {
251  caretTimer.resetSpan();
252  caretVisible = !caretVisible;
253  }
254  debugReplOutputStream.str("");
255  debugReplOutputStream << " Repl " << std::endl
256  << "==============" << std::endl
257  << debugReplAnsStream.str() << std::endl;
258 
259  debugReplOutputStream << std::endl
260  << "#;" << std::setfill('0') << std::setw(3) << debugReplLinum << "> "
261  << ofGetTextInput()
262  << (ofIsInputtingText() ? (caretVisible ? caret : ' ') : ' ');
263  }
264 }
