98  void load() {
99  // Enable Vertex Array so we can save our buffer, shader and
100  // attribute configurations on it
101  vao.init();
102  vao.bind();
103 
104  // Pass vertex data to vertex buffer
105  vbo.init();
106  vbo.setData(sizeof(vertices), vertices, ofBufferStaticDraw);
107 
108  // Loads shaders from file
109  std::string vs_src = ofLoadText("./res/vs.glsl"),
110  fs_src = ofLoadText("./res/fs.glsl");
111 
112  // Load and compile vertex shader
113  vertexShader.init(ofShaderVertex);
114  vertexShader.setSource(vs_src.c_str());
115  vertexShader.compile();
116  // Load and compile fragment shader
117  fragmentShader.init(ofShaderFragment);
118  fragmentShader.setSource(fs_src.c_str());
119  fragmentShader.compile();
120  // Attach shaders to program, unload them,
121  // and link the program
122  shaderProgram.init();
123  shaderProgram.attachUnload(vertexShader);
124  shaderProgram.attachUnload(fragmentShader);
125  shaderProgram.bindFragmentDataLocation("outColor");
126  shaderProgram.link();
127  // Use the program for the next operations on the VAO
128  shaderProgram.use();
129 
130 
131  // Load kitten texture
132  kitten = ofTexturePool::load("./res/sample.png");
133  puppy = ofTexturePool::load("./res/sample2.png");
134 
135 
136  // Setup position input attribute
137  attrPosition = shaderProgram.getAttributeLocation("position");
138  attrPosition.enable();
139  attrPosition.setProps(2, ofDataFloat, 7 * sizeof(float));
140  attrPosition.bindVertexArrayData(0);
141 
142  // Setup color input attribute
143  attrColor = shaderProgram.getAttributeLocation("color");
144  attrColor.enable();
145  attrColor.setProps(3, ofDataFloat, 7 * sizeof(float));
146  attrColor.bindVertexArrayData((void*)(2 * sizeof(float)));
147 
148  // Setup texel coordinate input attribute
149  attrTexcoord = shaderProgram.getAttributeLocation("texcoord");
150  attrTexcoord.enable();
151  attrTexcoord.setProps(2, ofDataFloat, 7 * sizeof(float));
152  attrTexcoord.bindVertexArrayData((void*)(5 * sizeof(float)));
153 
154  // Get location of Alpha uniform
155  variation = 1.0f;
156  uniVariation = shaderProgram.getUniformLocation("variation");
157  uniTexKitten = shaderProgram.getUniformLocation("texKitten");
158  uniTexPuppy = shaderProgram.getUniformLocation("texPuppy");
159  uniMVP = shaderProgram.getUniformLocation("mvp");
160 
161  // Set texture units for kitten and puppy
162  uniTexKitten.set(0);
163  uniTexPuppy.set(1);
164 
165  // Create element buffer so we can specify
166  // the vertex draw order
167  ebo.init();
168  ebo.setData(sizeof(elements), elements, ofBufferStaticDraw);
169  //ebo.setProps(3, ofDataUInt);
170  ebo.setProps(6, ofDataUInt);

172

174  // Look from... somewhere

197  void update(float dt) {
198  vao.bind();
199  // Make texture vary between 0 ~ 1 using a
200  // senoidal transformation
201  variation = 0.5f + (sin(lifetime.yieldSpan() * 0.5f) / 2.0f);
202  //ofLog(ofLogInfo, "Triangle ALPHA: %f\n", triangleAlpha);
203 
204  // Pass variation to shader program
205  uniVariation.set(variation);
206 
207 
208 
209  // Change rotation after pressing A
210  if(ofButtonTap(ofPadA)) {
211  rotationDirection *= -1.0f;
212  }
213  else if(ofButtonTap(ofPadLT)) {
214  rotationDirection = -1.0f;
215  }
216  else if(ofButtonTap(ofPadRT)) {
217  rotationDirection = 1.0f;
218  }
219  else if(ofButtonTap(ofPadY)) {
220  cameraPerspective = !cameraPerspective;
221  if(cameraPerspective) {
222  view = glm::lookAt(
223  glm::vec3(1.2f, 1.2f, -1.2f),
224  glm::vec3(0.0f, 0.0f, 0.0f),
225  glm::vec3(0.0f, 0.0f, -1.0f));
226  }
227  else {
228  // Look from above
229  view = glm::lookAt(
230  glm::vec3(0.0f, 0.0f, 1.2f),
231  glm::vec3(0.0f, 0.0f, 0.0f),
232  glm::vec3(0.0f, -1.0f, 0.0f));
233  }
234  }
235 
236  // Rotate our image!
237  // Transform so we rotate about 180 degrees per second
238  translation = glm::translate(translation, glm::vec3((ofGetLeftStick() * (-0.8f * dt)), 0.0f));
239  rotation = glm::rotate(rotation, glm::radians(90.0f * rotationDirection) * dt, glm::vec3(0.0f, 0.0f, 1.0f));
240 
241  // ORDER MATTERS HERE OKAY
242  model = translation * rotation;
243 
244  // Change projection angle based on up/down on right stick
245  float rsticky = ofGetRightStick().y;
246  if(rsticky != 0.0f) {
247  projectionAngle += 20.0f * rsticky * dt;
248  ofLog(ofLogInfo, "Projection Angle: %0.2f\n", projectionAngle);
249  }
250  projectionAngle = ofClamp(projectionAngle, 30.0f, 90.0f);
251  if(cameraPerspective)
252  projection = glm::perspective(glm::radians(projectionAngle), // Field of view
253  800.0f / 600.0f, // Aspect Ratio
254  1.0f, 10.0f); // Near/Far
255  else projection = glm::ortho(-1.0f, 1.0f, 1.0f, -1.0f, 1.0f, 10.0f);
256 
257 
258  // Dispatch matrices
259  uniMVP.set(projection * view * model);
260 
261  if(ofStickMovedTowards(ofStickLeft | ofStickHoriz | ofStickNegative, ofPlayerTwo)
262  || ofStickMovedTowards(ofStickLeft | ofStickHoriz | ofStickPositive, ofPlayerTwo)
263  || ofStickMovedTowards(ofStickLeft | ofStickVert | ofStickNegative, ofPlayerTwo)
264  || ofStickMovedTowards(ofStickLeft | ofStickVert | ofStickPositive, ofPlayerTwo)) {
265  glm::vec2 p2lstick = ofGetLeftStick(ofPlayerTwo);
266  ofLog(ofLogInfo, "Player 2 left stick tap: {%0.2f, %0.2f}\n", p2lstick.x, p2lstick.y);
267  }
268 
269  if(ofButtonTap(ofPadB)) {
270  ofScmEval("(display \"Button B pressed; informed through Scheme REPL\\n\")");
271  }
272 
273  if(ofButtonTap(ofPadBack) || ofButtonTap(ofPadBack, ofPlayerTwo))
274  ofSoftStop();
275 
276  // Benchmarking
277  //ofBenchmarkUpdateCall();
278  vao.unbind();
279  }
